const requestDeduplication = () => {

        // å° GET è«‹æ±‚é€²è¡Œå»é‡

                const pending = pendingRequests.get(key);

                        reject(error);
                        pendingRequests.delete(key);

                // è¨­ç½®è¶…æ™‚
                setTimeout(() => {
                    if (pendingRequests.has(key)) {
                        pendingRequests.delete(key);
                        reject(new Error('è«‹æ±‚è¶…æ™‚'));
                    }

    };
